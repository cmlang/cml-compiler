class Book(Product, ABC):

    def __init__(self, title: 'str', description: 'str') -> 'None':
        super().__init__(description)
        self.__title = title

    @property
    def title(self) -> 'str':
        return self.__title

    def __str__(self) -> 'str':
        return "%s(title=%s, description=%s)" % (
            type(self).__name__,
            self.title,
            self.description
        )
