class Book(Product, StockItem, ABC):

    @abstractproperty
    def title(self) -> 'str':
        pass

    @staticmethod
    def create_book(title: 'str', base_property: 'str') -> 'Book':
        base = Base.extend_base(base_property)
        product = Product.extend_product(base)
        stock_item = StockItem.extend_stock_item(base)
        return BookImpl(base, product, stock_item, title)

    @staticmethod
    def extend_book(base: 'Base', product: 'Product', stock_item: 'StockItem', title: 'str') -> 'Book':
        return BookImpl(base, product, stock_item, title)
