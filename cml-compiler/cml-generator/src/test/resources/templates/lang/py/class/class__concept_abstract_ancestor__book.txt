class Book(Product, ABC):

    def __init__(self, title: 'str', description: 'str') -> 'None':
        super().__init__(description, 0)
        self.__title = title

    @property
    def title(self) -> 'str':
        return self.__title

    @property
    def name(self) -> 'str':
        return self.title

    @property
    def value(self) -> 'Decimal':
        return Decimal("3")

    def __str__(self) -> 'str':
        return "%s(title=%s, name=%s, value=%s, description=%s)" % (
            type(self).__name__,
            self.title,
            self.name,
            self.value,
            self.description
        )
