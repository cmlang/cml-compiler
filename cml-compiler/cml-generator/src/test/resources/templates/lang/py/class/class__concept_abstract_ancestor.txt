class Book(Product, ABC):

    def __init__(self, title: 'str', name: 'str', description: 'str') -> 'None':
        super().__init__(name, description)
        self.__title = title
        self.__name = name

    @property
    def title(self) -> 'str':
        return self.__title

    @property
    def name(self) -> 'str':
        return self.__name

    def __str__(self) -> 'str':
        return "%s(title=%s, name=%s, description=%s)" % (
            type(self).__name__,
            self.title,
            self.name,
            self.description
        )
