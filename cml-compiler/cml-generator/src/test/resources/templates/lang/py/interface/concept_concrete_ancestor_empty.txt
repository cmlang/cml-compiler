class Book(Intermediate, ABC):

    @staticmethod
    def create_book(description: 'str') -> 'Book':
        product = Product.extend_product(description)
        intermediate = Intermediate.extend_intermediate(product)
        return BookImpl(product, intermediate)

    @staticmethod
    def extend_book(product: 'Product', intermediate: 'Intermediate') -> 'Book':
        return BookImpl(product, intermediate)
