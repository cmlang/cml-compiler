class Book(ABC):

    @abstractproperty
    def title(self) -> 'str':
        pass

    @abstractproperty
    def sequel(self) -> 'Book':
        pass

    @abstractproperty
    def categories(self) -> 'List[Category]':
        pass

    @staticmethod
    def create_book(title: 'str', sequel: 'Book', categories: 'List[Category]') -> 'Book':
        return BookImpl(title, sequel, categories)

    @staticmethod
    def extend_book(title: 'str', sequel: 'Book', categories: 'List[Category]') -> 'Book':
        return BookImpl(title, sequel, categories)
