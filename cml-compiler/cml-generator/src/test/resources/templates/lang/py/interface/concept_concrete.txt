class Book(ABC):

    @abstractproperty
    def title(self) -> 'str':
        pass
    @abstractproperty
    def sequel(self) -> 'Book':
        pass
    @abstractproperty
    def categories(self) -> 'Set[Category]':
        pass

    @staticmethod
    def create_book(title: 'str', sequel: 'Book', categories: 'Set[Category]') -> 'Book':
        return BookImpl(title, sequel, categories)
