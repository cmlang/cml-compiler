class Book(Product, ABC):

    @abstractproperty
    def title(self) -> 'str':
        pass

    @staticmethod
    def create_book(title: 'str') -> 'Book':
        product = Product.extend_product()
        return BookImpl(product, title)

    @staticmethod
    def extend_book(product: 'Product', title: 'str') -> 'Book':
        return BookImpl(product, title)
