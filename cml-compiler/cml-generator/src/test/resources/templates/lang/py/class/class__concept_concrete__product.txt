class Product(ABC):

    def __init__(self, description: 'str', value: 'Decimal') -> 'None':
        self.__description = description
        self.__value = value

    @property
    def description(self) -> 'str':
        return self.__description

    @property
    def value(self) -> 'Decimal':
        return self.__value

    @abstractproperty
    def name(self) -> 'str':
        pass

    def __str__(self) -> 'str':
        return "%s(name=%s, description=%s, value=%s)" % (
            type(self).__name__,
            self.name,
            self.description,
            self.value
        )
