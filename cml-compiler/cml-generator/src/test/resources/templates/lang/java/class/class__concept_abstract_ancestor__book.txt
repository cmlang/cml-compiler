abstract class Book extends Product
{
    private final String title;

    public Book(String title, String description)
    {
        super(description, 0);
        this.title = title;
    }

    public String getTitle()
    {
        return this.title;
    }

    public String getName()
    {
        return getTitle();
    }

    public BigDecimal getValue()
    {
        return new BigDecimal("3");
    }

    public String toString()
    {
        return new StringBuilder(Book.class.getSimpleName())
                   .append('(')
                   .append("title=").append(String.format("\"%s\"", getTitle())).append(", ")
                   .append("name=").append(String.format("\"%s\"", getName())).append(", ")
                   .append("value=").append(String.format("\"%s\"", getValue())).append(", ")
                   .append("description=").append(String.format("\"%s\"", getDescription()))
                   .append(')')
                   .toString();
    }
}
