interface Book extends Product, StockItem
{
    String getTitle();

    static Book createBook(String title, String baseProperty)
    {
        Base base = Base.extendBase(baseProperty);
        Product product = Product.extendProduct(base);
        StockItem stockItem = StockItem.extendStockItem(base);
        return new BookImpl(base, product, stockItem, title);
    }

    static Book extendBook(Base base, Product product, StockItem stockItem, String title)
    {
        return new BookImpl(base, product, stockItem, title);
    }
}
