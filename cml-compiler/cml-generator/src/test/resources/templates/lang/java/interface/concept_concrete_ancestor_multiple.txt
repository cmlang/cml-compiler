interface Book extends Product, StockItem
{
    String getTitle();

    BigDecimal getValue();

    static Book createBook(String title, String baseProperty)
    {
        return createBook(title, new BigDecimal("10.0"), baseProperty);
    }

    static Book createBook(String title, BigDecimal value, String baseProperty)
    {
        Base base = Base.extendBase(baseProperty);
        Product product = Product.extendProduct(base);
        StockItem stockItem = StockItem.extendStockItem(base);
        return new BookImpl(base, product, stockItem, title, value);
    }

    static Book extendBook(Base base, Product product, StockItem stockItem, String title, BigDecimal value)
    {
        return new BookImpl(base, product, stockItem, title, value);
    }
}
