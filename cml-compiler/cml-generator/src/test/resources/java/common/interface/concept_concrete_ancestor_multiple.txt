interface Book extends Product, StockItem
{
    String getTitle();

    static Book createBook(String baseProperty, String title)
    {
        Base base = Base.createBaseImpl(baseProperty);
        Product product = Product.createProductImpl(base);
        StockItem stockItem = StockItem.createStockItemImpl(base);
        return new BookImpl(base, product, stockItem, title);
    }

    static Book createBookImpl(Base base, Product product, StockItem stockItem, String title)
    {
        return new BookImpl(base, product, stockItem, title);
    }
}
