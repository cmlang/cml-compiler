class Book:

    def __init__(self, title: str, sequel: Book) -> None:
        self.__title = title
        self.__sequel = sequel

    @property
    def title(self) -> str:
        return self.__title

    @property
    def sequel(self) -> Book:
        return self.__sequel

    def __str__(self) -> str:
        return "%s(title=%s, sequel=%s)" % (
            type(self).__name__,
            self.title,
            self.sequel
        )
