class BookImpl(Book):

    def __init__(self, description: str):
        product = Product.create(description);
        intermediate = Intermediate.create(product);

    def __init__(self, product: Product, intermediate: Intermediate):
        self.__product = product
        self.__intermediate = intermediate

    def description(self) -> str:
        return self.__product.description

    def __str__(self) -> str:
        return "%s(description=%s)" % (
            type(self).__name__,
            self.description
        )
