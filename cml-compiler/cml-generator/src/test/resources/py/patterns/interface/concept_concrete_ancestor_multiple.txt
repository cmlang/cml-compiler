class Book(ABC, Product, StockItem):

    @abstractproperty
    def title(self) -> String:
        pass

    static Book create(String base_property, String title)
    {
        return new BookImpl(base_property, title);
    }

    static Book create(Base base, Product product, StockItem stock_item, String title)
    {
        return new BookImpl(base, product, stock_item, title);
    }
