class Book(ABC):

    @abstractproperty
    def title(self) -> String:
        pass
    @abstractproperty
    def sequel(self) -> Book:
        pass
    @abstractproperty
    def categories(self) -> Set[Category]:
        pass

    static Book create(String title, Book sequel, Set[Category] categories)
    {
        return new BookImpl(title, sequel, categories);
    }
