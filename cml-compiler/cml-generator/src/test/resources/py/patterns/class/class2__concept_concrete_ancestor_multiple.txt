class BookImpl(Book):

    def __init__(self, base_property: String, description: String, quantity: Integer, title: String):
        base = Base.create(base_property);
        product = Product.create(base, description);
        stock_item = StockItem.create(base, quantity);

        self.__title = title

    def __init__(self, base: Base, product: Product, stock_item: StockItem, title: String):
        self.__base = base
        self.__product = product
        self.__stock_item = stock_item
        self.__title = title

    @property
    def title(self) -> String:
        return self.__title

    def base_property(self) -> String:
        return self.__base.base_property

    def description(self) -> String:
        return self.__product.description

    def quantity(self) -> Integer:
        return self.__stock_item.quantity

    def __str__(self) -> str:
        return "%s(base_property=%s, description=%s, quantity=%s, title=%s)" % (
            type(self).__name__,
            self.base_property,
            self.description,
            self.quantity,
            self.title
        )
