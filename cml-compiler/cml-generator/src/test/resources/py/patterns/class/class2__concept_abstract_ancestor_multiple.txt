class BookImpl(ABC, Book):

    def __init__(self, base: Base, product: Product, stock_item: StockItem, title: String):
        self.__base = base
        self.__product = product
        self.__stock_item = stock_item
        self.__title = title

    @property
    def title(self) -> String:
        return self.__title

    def base_property(self) -> String:
        return self.__base.base_property

    def description(self) -> String:
        return self.__product.description

    def quantity(self) -> Integer:
        return self.__stock_item.quantity

    def __str__(self) -> str:
        return "%s(base_property=%s, description=%s, quantity=%s, title=%s)" % (
            type(self).__name__,
            self.base_property,
            self.description,
            self.quantity,
            self.title
        )
